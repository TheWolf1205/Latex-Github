\begin{homeworkProblem}
  Verifique se as afirmaûÏûçes sûÈo verdadeiras ou falsas, com justificativa ($G$ sempre denotarûÀ um grupo).
  \begin{enumerate}
    \item Se $x,y \in G$ tûˆm ordem finita entûÈo $xy$ tem ordem finita.
    
    \item Se $x$ e $y$ sûÈo elementos de ordem finita que comutam em $G$ entûÈo $xy$ tem ordem finita.
    
    \item Se todo elemento de $G$ tem ordem finita entûÈo $G$ tem ordem finita.
    
    \item Se $G$ tem ordem finita entûÈo existe $m \in \mathbb{N}$ tal que $g^m = 1$, $\forall g \in G$ e alûˋm disso, existe $a \in G$ tal que $o(a)=m$.
    
    \item Se $G$ ûˋ abeliano de ordem finita entûÈo existe $m \in \mathbb{N}$ tal que $g^m = 1$, $\forall g \in G$ e alûˋm disso,existe $a \in G$ tal que $o(a)=m$.
  \end{enumerate}
  \begin{solution}
    \begin{enumerate}
      \item Se $x,y \in G$ tûˆm ordem finita entûÈo $xy$ tem ordem finita.

        Falsa.\\
        Seja $G=Gl_2(\mathbb{R})$, logo considere $x$ e $y$ como  
        \begin{align*}
          x=\begin{pmatrix}
            -1 & 0 \\
            0 & 1
          \end{pmatrix},\qquad y=\begin{pmatrix}
            1 & 1 \\
            0 & -1
          \end{pmatrix}.
        \end{align*}
        Pode-se calcular que $o(x)=o(y)=2$, mais
        \begin{align*}
          xy&=\begin{pmatrix}
            -1 & -1 \\
            0 & -1
          \end{pmatrix}\\
          (xy)^2&=\begin{pmatrix}
            1 & 2 \\
            0& 1
          \end{pmatrix}\\
          (xy)^{k}&=\begin{pmatrix}
            (-1)^{k} & (-1)^{k}k \\
            0 & (-1)^{k}
          \end{pmatrix},
        \end{align*}
        i,e, $o(xy)=\infty$.
      \item Se $x$ e $y$ sûÈo elementos de ordem finita que comutam em $G$ entûÈo $xy$ tem ordem finita.

        Verdadeira.\\
        Como $x$ e $y$ sûÈo elementos de orden finita que comutam en $G$, suponha que $o(x)=h$ e $o(y)=k$, entûÈo $o(xy)=\MMC(h,k)$ dado que 
        \begin{align*}
          \underbrace{xy*xy*\cdots*xy}_{\MMC(h,k)\text{ vezes}}&=(\underbrace{x*x*\cdots*x}_{\MMC(h,k) \text{ vezes}})*(\underbrace{y*y*\cdots*y}_{\MMC(h,k) \text{ vezes}})\\
          &=1*1\\
          &=1.
        \end{align*}
      \item Se todo elemento de $G$ tem ordem finita entûÈo $G$ tem ordem finita.

      Falsa.\\
      Considere $G=(\mathbb{Q}/\mathbb{Z},+)$, entûÈo sim $a,b\in \mathbb{Z}$, $x=\frac{a}{b}+\mathbb{Z}\in \mathbb{Q}/\mathbb{Z}$, logo observe que
      \begin{align*}
        \underbrace{\frac{a}{b}+\mathbb{Z}+\cdots+\frac{a}{b}+\mathbb{Z}}_{b\text{ vezes}}=\frac{ab}{b}+\mathbb{Z}=a+\mathbb{Z}=\mathbb{Z}. 
      \end{align*}
      Portanto, podemos afirmar que $o(x)<b<\infty$ para tudo $x\in G$, pero $|G|=\infty$.
      \item Se $G$ tem ordem finita entûÈo existe $m \in \mathbb{N}$ tal que $g^m = 1$, $\forall g \in G$ e alûˋm disso, existe $a \in G$ tal que $o(a)=m$.

        Falsa.\\
        Tome como exemplo $S_3$. 
      \item Se $G$ ûˋ abeliano de ordem finita entûÈo existe $m \in \mathbb{N}$ tal que $g^m = 1$, $\forall g \in G$ e alûˋm disso, existe $a \in G$ tal que $o(a)=m$.

      Verdadeira.\\
      Como $G$ ûˋ um grupo, satisfaze Fechadura, entûÈo como $|G|<\infty$ se cumple que para tudo $g\in G, o(g)<\infty$, entûÈo tome $\displaystyle m=\MMC_{h\in G}(o(h))$, logo $g^{m}=1$ para tudo $g\in G$.\\
      Logo tome $\displaystyle x=\prod_{h\in G}h$, entûÈo $o(x)=m$.
    \end{enumerate}
  \end{solution}
\end{homeworkProblem}
