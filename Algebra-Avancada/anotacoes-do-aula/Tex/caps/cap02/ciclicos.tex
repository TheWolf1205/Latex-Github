Seja $G$ um grupo cíclico gerado por $g$, neste caso temos 2 opções
\begin{itemize}
  \item $G$ é infinito.\\
    Então definimos $\varphi:G\to(\mathbb{Z},+)$ por $\varphi(g^{k})=k$, logo $G_1\cong \mathbb{Z}$.
  \item $G$ é finito.\\
    Suponha $|G|=n$, então para $n\geq 2$ vamos considerar o grupo
    \begin{align*}
      \mathbb{Z}_{n}:=\{\overline{0},\overline{1},\cdots,\overline{n-1}\},\quad\text{das classe de vetor módulo $n$.}
    \end{align*}
    Logo $\varphi:G\to\mathbb{Z}_{n}$ definido por $\varphi(g^{k})=\overline{k}$ é um isomorfismo, por tanto $G\cong \mathbb{Z}_{n}$ 
\end{itemize}
