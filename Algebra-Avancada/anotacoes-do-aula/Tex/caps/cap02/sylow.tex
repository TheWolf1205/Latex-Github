\begin{theorem}[Teoremas de Silow]
  Seja $G$ grupo finito e $p$ primo com $p\mid |G|$, i.Ă©, $|G|=p^{\alpha}m$ com $\alpha\geq 1$ e $\mdc(p,m)=1$, entĂŁo
  \begin{enumerate}
    \item Para qualquer $1\leq \beta\leq \alpha$, existe um subgrupo de $G$ de ordem $p^{\beta}$.\\
      Los $p$-subgrupos de ordem $p^{\alpha}$ sĂŁo chamados os \textbf{$p$-subgrupos de Sylow} e denotamos $Syl_{p}(G)=\{H:H\text{ Ă© $p$-subgrupo de Sylow}\}$.
    \item Dado qualquer $p$-subgrupo $P$, existe $S\in Syl_{p}(G)$ tal que $P\leq S$. Consequentemente, todos os $p$-subgrupos de Sylow sĂŁo conjugados.
    \item Se $n_p$ Ă© a quantidade de $p$-subgrupos de Sylow de $G$ (ou seja, $n_p=|Syl_{p}(G)|$), entĂŁo $n_p\mid m$ e $n_p\equiv 1(mod\,p)$ 
  \end{enumerate}
\end{theorem}
\begin{example}
  Seja $G$ com $|G|=12=2^{2}\cdot 3$, entĂŁo existe $P\in Syl_{2}(G)$, com $|P|=2^{2}$ e existe $Q\in Syl_{3}(G)$, com $|G|=3$.\\
  Nos temos que
  \begin{itemize}
    \item $n_2=|Syl_{2}(G)|\in \{1,3\}$, ja que $n_2\mid 3$.\\
      AlĂ©m mais, $n_2\equiv 1(mod\, 2)$, logo
      \begin{align*}
        \begin{cases}
          n_2=1\equiv 1(mod\,2), &\text{ EntĂŁo $P\normal G$, ja que $xPx^{-1}=P$}\text{,} \\
          n_2=3\equiv 1(mod\,2), &.
        \end{cases}
      \end{align*}
    \item $n_3=|Syl_{3}(G)|\in \{1,2,4\}$, ja que $n_3\mid 2^{2}$.\\
      AlĂ©m mais, $n_3\equiv 1(mod\, 3)$, logo
      \begin{align*}
        \begin{cases}
          n_3=1\equiv 1(mod\,3), &\text{ EntĂŁo $Q\normal G$}\text{,} \\
          n_3=2\not\equiv 1(mod\,3), &\\
          n_3=4\equiv 1(mod\,3), &\text{ Observe o seguinte}
        \end{cases}
      \end{align*}
      Suponha que $n_p=4$, logo temos que se $Q_i\in Syl_{3}(G)$ com $i\in\{1,2,3,4\}$, logo $\bigcap_{i=1}^{4}Q_i=\{1\}$, pelo que temos $4\times 2=8$ elementos distintos de ordem $3$ em $G$. Logo como $|G|=12$, restam $4$ elementos que forma um Ăşnico subgrupo de ordĂ©m $4$ (isso pode ser visto usando um argumento de contagem), logo $P\normal G$. 
  \end{itemize}
\end{example}
