\begin{definition}[Grupo]
  Um Grupo $(G,*)$ ĂŠ um conjunto $G$ nĂŁo vazio com uma operaĂ§ĂŁo $*$ que satisfaz as seguintes propriedades
\begin{itemize}
  \item \textbf{Fechamento:} Se $a,b\in G$, entĂŁo $a*b\in G$.
  \item \textbf{Associatividade:} Para quaisquer $a,b,c\in G$, tem-se que $a*(b*c)=(a*b)*c$.
  \item \textbf{ExistĂŞncia do Elemento Identidade:} Existe um elemento $e\in G$ tal que $e*a=a=a*e$ para tudo $a\in G$.
  \item \textbf{ExistĂŞncia do Elemento Inverso:} Para cada $a\in G$, existe um elemento $b\in G$ tal que $a*b=e=b*a$. Geralmente o elemento identidade se denota por $e$ o $1$. 
\end{itemize}
\end{definition}
\begin{definition}[Grupo Abeliano]
  Um Grupo $(G,*)$ ĂŠ um grupo abeliano si ele ĂŠ Grupo y alem do mais, ele satisfaz
  \begin{itemize}
    \item \textbf{Comutatividade:} Para todo par $a,b\in G$, tem-se que $a*b=b*a$.
  \end{itemize}
\end{definition}

\begin{definition}[Ordem do grupo]
  Definimos ordem do grupo a o nĂşmero de elementos do conjunto $G$, denotado por $|G|$. 
\end{definition}
\begin{definition}[Ordem do elemento]
  A ordem de um elemento $a\in G$ ĂŠ definida como o menor enteiro positivo $n$ tal que $a^{n}=\underbrace{a\cdot a\cdots a}_{n \text{ vezes}}=1$ e denotada por $o(a)=n$. Se nĂŁo existe tal $n$, dizemos que $a$ tem ordem infinita e denotamos $o(a)=\infty$. 
\end{definition}
