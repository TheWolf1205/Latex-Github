\begin{homeworkProblem}
  Seja $G$ um grupo de ordem par. Mostre que $G$ tem um elemento $g \neq 1$ tal que $g^2 = 1$.
  \begin{solution}
    Como $G$ é um grupo de ordem par, existe $h\in G$ tal que pra um enteiro $k$ se satisfaze que $h^{2k}=1$, logo si fixamos $g=h^{k}$, então $g^{2}=1$.  
  \end{solution}
\end{homeworkProblem}
