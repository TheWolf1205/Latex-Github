  \begin{align*}
    f(u)=\sum_{n=1}^{\infty}\frac{1}{2^{n}}u_{n}.
  \end{align*}
  \begin{enumerate}[(i)]
    \item Muestre que $f\in E^{*}$ y calcule $\norm{f}_{E^{*}}$.
    \item ¿Es posible encontrar $u\in E$ tal que $\norm{u}=1$ y $f(u)=\norm{f}_{E^{*}}$?
