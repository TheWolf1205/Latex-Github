\begin{theorem}[Teorema de aplicaĂ§ĂŁo inversa em uma variĂĄvel]
  Seja $f:I\subseteq \mathbb{R}\to\mathbb{R}$ derivĂĄvel e $f^{\prime}(x)\neq 0$ para todo $x\in I$, entĂŁo
  \begin{itemize}
    \item $f$ ĂŠ monĂłtona (logo injetiva) (pelo Teorema de Darboux). 
    \item $f(I)=J$ ĂŠ um intervalo aberto, alĂŠm disso $g=f^{-1}:J\to I$ ĂŠ diferenciĂĄvel e $g^{\prime}(f(x))=\left[ f(x) \right]^{-1}$ ou seja $g$ ĂŠ a inversa global. 
  \end{itemize}
\end{theorem}
O que podemos generalizar para $\mathbb{R}^{m}$?

$f^{\prime}(x)\neq 0$ (em $\mathbb{R}^{m}$) para todo $x\in\mathbb{R}^{m}$ somente sabesmos que $f^{\prime}(x)$ e isomorfismo (linear e bijetiva), pelo nĂŁo podemos garantir inversa global, so local.\\
Pode tomar como exemplo $f(x,y)=(e^{x}\cos(y),e^{x}\sen(y))$, mas veremos isso mĂĄs tarde.
\begin{definition}[Difeomorfismo]
  Seja $U\subseteq \mathbb{R}^{m}$, $V\in\mathbb{R}^{n}$ abertos, $f:U\to V$ ĂŠ um difeomorfismo quando:
  \begin{enumerate}
    \item $f$ ĂŠ bijetiva.
    \item $f$ ĂŠ diferenciĂĄvel.
    \item $g=f^{-1}:V\to U$ ĂŠ diferenciĂĄvel. 
  \end{enumerate}
  Se $f,g\in C^{k}$  dizemos que $f$ ĂŠ $C^{k}$-difeomorfismo. 
\end{definition}
\begin{proposition}\label{prop:difeo-identidade}
  Se $f:U\to V$ ĂŠ um difeomorfismo, entĂŁo para todo $x\in U$.
  \begin{align*}
    g^{\prime}(f(x))\cdot f^{\prime}(x)=Id_{\mathbb{R}^{m}},\quad f^{\prime}(g(y))\cdot g^{\prime}(y)=Id_{\mathbb{R}^{n}}.
  \end{align*}
\end{proposition}
\begin{proof} 
  Sabemos que $g(f(x))=x$ e $f(g(y))=y$ com $x\in U$ e $y\in V$, entĂŁo, pela regra da cadeia sabemos que
  \begin{align*}
    \left( g\circ f \right)^{\prime}(x)=g^{\prime}(f(x))\cdot f^{\prime}(x)=\left( Id_{U} \right)^{\prime}(x)=Id_{\mathbb{R}^{m}}.\\
    \left( f\circ g \right)^{\prime}(y)f^{\prime}(g(y))\cdot g^{\prime}(y)=\left( Id_{V} \right)^{\prime}(y)=Id_{\mathbb{R}^{n}}.
  \end{align*}
\end{proof}
\begin{proposition}
  Se $f:U\to V$ ĂŠ difeomorfismo, entĂŁo $m=n$. 
\end{proposition}
\begin{proof} 
  Pelo proposiĂ§ĂŁo \ref{prop:difeo-identidade}, $f^{\prime}(x)$ ĂŠ um isomorfismo linear, entĂŁo $f^{\prime}(x)$ ĂŠ bijetiva y leva elementos de uma base de $U$ em elementos de uma base de $V$, logo $m=n$.
\end{proof}
\begin{definition}[Difeomorfismo local]
  Uma aplicaĂ§ĂŁo diferenciĂĄvel $f:U\mathbb{R}^{n}\to\mathbb{R}^{n}$ ĂŠ um \textbf{difeomorfismo local} quando para cada $x\in U$, existe uma bola aberta $B_x\subseteq U$ contendo $x$ tal que $f\big|_{B_x}$ ĂŠ difeomorfismo. 
\end{definition}
\begin{proposition}
  Se $f:U\to\mathbb{R}^{n}$ ĂŠ um difeomorfismo local, entĂŁo $f$ leva abertos em abertos.
\end{proposition}
\begin{proof} 
  Seja $A\subseteq U$ aberto. Para cada $x\in A$, existe $B_x\subseteq A$ aberta tal que $f\big|_{B_x}$ ĂŠ difeomorfismo, logo $A=\bigcup_{x\in A}B_x$ e $f(B_x)$ ĂŠ aberto para todo $x\in A$. Portanto,
  \begin{align*}
    f(A)=f\left( \bigcup_{x\in A}B_x \right)=\bigcup_{x\in A}f(B_x).
  \end{align*}
  Assim, $f(A)$ ĂŠ aberto. 
\end{proof}
\begin{proposition}
  Se $f:U\to\mathbb{R}^{m}$ ĂŠ um difeomorfismo local injetivo, entĂŁo $f:U\to f(U)$ ĂŠ um difeomorfismo global. 
\end{proposition}
\begin{proof} 
  $f$ ĂŠ injetiva, entĂŁo $f$ ĂŠ bijetiva sobre $f(U)$. Logo para cada $y\in f(U)$, existe $x\in U$ tal que $f^{-1}(y)=x$, mas como $f$ ĂŠ difeomorfismo local, existe $B_x$ tal que $f\big|_{B_x}$ ĂŠ difeomorfismo. EntĂŁo $f(B_x)$ ĂŠ uma vizinhanca de $y$, onde a inversa local coincide com $f^{-1}(y)$, logo como $x$ ĂŠ arbitrario, entĂĄo $f^{-1}$ ĂŠ global y diferenciĂĄvel em cada ponto $y$, por tanto $f$ ĂŠ um difeomorfismo. 
\end{proof}

  \end{align*}
  \textbf{AfirmaĂ§ĂŁo:} $f$ ĂŠ um difeomorfismo entre $\mathbb{R}^{m}$ e $B$ (a bola unitaria). 
\end{example}

\begin{lemma}\label{lema:inversa-e-cinfinito}
  A aplicaĂ§ĂŁo
  \begin{align*}
    inv:GL(\mathbb{R}^{m})&\to GL(\mathbb{R}^{m})\\
    x&\to x^{-1},
  \end{align*}
  ĂŠ $C^{\infty}$.
\end{lemma}
Lembremos que
\begin{itemize}
  \item $GL(\mathbb{R}^{m})=\{x\in \mathbb{R}^{m\times m}:\det(x)\neq 0\}$.
  \item $GL(\mathbb{R}^{m})$ ĂŠ aberto de $\mathbb{R}^{m\times m}$, poiss a funĂ§ĂŁo
    \begin{align*}
      \det:\mathbb{R}^{m\times m}&\to\mathbb{R}\\
      x&=\det(x)
    \end{align*}
    ĂŠ contĂ­nua (ĂŠ um polinomio nas entradas).\\
    Logo $\det(\mathbb{R}\setminus\{0\})=GL(\mathbb{R}^{n})$.
\end{itemize}
\begin{proof} 
  Sabemos que $x^{-1}=\frac{1}{\det(x)}\cdot Adj(x)$, logo $x^{-1}$ ĂŠ uma funĂ§ĂŁo racional com $\det(x)\neq 0$, assim $inv\in C^{\infty}$. 
\end{proof}
\begin{lemma}\label{lema:2-ckdifeomorfismo}
  Se $f:U\to V$ ĂŠ um $C^{k}$-difeomorfismo, entĂŁo para todo $y=f(x)\in V$, temos que $g^{\prime}(y)=\left[ f^{\prime}(x) \right]^{-1}=\left[ f^{\prime}\left( g(y) \right) \right]^{-1}$.
\end{lemma}
\begin{proof} 
  Como $g(f(x))=x$, para todo $x\in U$, pela regra da cadeia temos que, $g^{\prime}(f(x))\cdot f^{\prime}(x)=Id_{\mathbb{R}^{m}}$. EntĂŁo $g^{\prime}(y)=g^{\prime}\left( f(x) \right)=\left[ f^{\prime}(x) \right]^{-1}=\left[ f^{\prime}(g(y)) \right]^{-1}$.
\end{proof}
\begin{theorem}\label{teo:3.6-teorema1}
  Se $f:U\to V$ ĂŠ um $C^{k}$-difeomorfismo com $U\subseteq \mathbb{R}^{m}$ e $V\subseteq\mathbb{R}^{n}$ abertos, entĂŁo $g=f^{-1}:V\to U$ tambĂŠm ĂŠ $C^{k}$. 
\end{theorem}
\begin{proof} 
  Por induĂ§ĂŁo sobre $k$.\\
  $k=1$.
  \begin{itemize}
    \item $f\in C^{1}$, diferenciĂĄvel e $f^{\prime}$ ĂŠ contĂ­nua.
    \item $g$ ĂŠ diferenciĂĄvel, entĂŁo $g$ ĂŠ contĂ­nua. 
  \end{itemize}
  Ja que $g^{\prime}=inv\circ f^{\prime}\circ g$, entĂŁo $g^{\prime}$ ĂŠ contĂ­nua (al ser composta de contĂ­nuas) e portanto $g\in C^{1}$.

  Paso indutivo,.\\
  Sumponha que o resultado ĂŠ certo para $k-1$, vamos verificar o resultado para $k$.
  \begin{itemize}
    \item $f^{\prime}\in C^{k-1}$ (pois $f\in C^{k}$).
    \item $g\in C^{k-1}$ (pois $f\in C^{k-1}$).
    \item $inv\in C^{k-1}$ (pelo lema \ref{lema:2-ckdifeomorfismo}).
  \end{itemize}
  Como $g^{\prime}=inv\circ f^{\prime}\circ g$, entĂŁo $g^{\prime}\in C^{k-1}$, i.e, $g\in C^{k}$.
\end{proof}
\begin{theorem}\label{teo:3.6-teorema2}
  Seja $f:U\subseteq \mathbb{R}^{m}\to\mathbb{R}^{n}$ de classe $C^{1}$, com $a\in U$. Se $f^{\prime}(a):\mathbb{R}^{m}\to\mathbb{R}^{n}$ ĂŠ injetiva, entĂŁo existem $\delta>0$ e $c>0$ taes que $|f(x)-f(y)|\geq c|x-y|$ para todo $x,y\in B(a,\delta)$. 
\end{theorem}
\begin{proof} 
  Como $f^{\prime}(a)$ ĂŠ injetiva $f^{\prime}(a)\cdot v\neq 0$ para todo $v\in\mathbb{R}^{m}$.\\
  Defina a funĂ§ĂŁo $\phi(u)=|f^{\prime}(a)\cdot u|$ para todo $u\in S^{m-1}$. Note que $\phi$ ĂŠ contĂ­nua. Assim, pelo teorema de Weierstrass, $\phi$ atingue um mĂ­nimo. Este mĂ­nimo ĂŠ positivo pois $\phi(u)>0$ para todo $u\in S^{m-1}$. EntĂŁo, existe $c>0$ tal que
  \begin{align*}
    |f^{\prime}(a)\cdot u|\geq 2>0.
  \end{align*}
  Logo
  \begin{align*}
    |f^{\prime}(a)\cdot \frac{v}{|v|}||v|\geq 2c|v|>0,\quad\text{ para todo }v\in\mathbb{R}^{m}.
  \end{align*}
  Agora, defina $r(x)=f(x)-f(a)-f^{\prime}(a)(x-a)$. EntĂŁo
  \begin{itemize}
    \item $r(a)=0$.
    \item $r^{\prime}(a)=f^{\prime}(a)-f^{\prime}(a)=0$.
  \end{itemize}
  Como $r^{\prime}$ ĂŠ contĂ­nua em $a$, temos que dado $\epsilon=c>0$, existe $\delta^{\prime}>0$ tal que se $|x-a|<\delta^{\prime}$, entĂŁo $|r^{\prime}(x)-r^{\prime}(a)|=|r^{\prime}(x)|<\epsilon=c$.\\
  Seja $B=B(a,\delta)$ com $\delta=\frac{\delta^{\prime}}{2}$, logo dados $x,y\in B$, temos que $[x,y]\in B$. Logo pela desigualdade do valor mĂŠdio, temos que
  \begin{align*}
    |r(x)-r(y)|\leq c|x-y|.
  \end{align*}
  Assim
  \begin{align*}
    |f(x)-f(y)|\geq|f^{\prime}(a)(x-y)|-|r(x)-r(y)|\geq 2c|x-y|-c|x-y|\geq c|x-y|.
  \end{align*}
\end{proof}
\begin{theorem}[Difeomorfismo do homomorfismo inverso]\label{teo:3.6-teorema-difeo-inverso}
  Seja $f:U\to V$ um homeomorfismo diferenciĂĄvel entre abertos $U,V\in\mathbb{R}^{n}$. Se para algĂşm $x\in U$, $f^{\prime}(x)$ ĂŠ invertivel, entĂŁo $g=f^{-1}$ ĂŠ diferenciĂĄvel em $y=f(x)$ e $g^{\prime}(y)=\left[ f^{\prime}(x) \right]^{-1}$. 
\end{theorem}
\begin{proof} 
  Sejam $y=f(x)\in V$. Para $v\in\mathbb{R}^{m}$ tal que $x+v\in U$, defina
  \begin{align*}
    w=f(x+y)-f(x)=f(g(y)-v)-y.
  \end{align*}
  EntĂŁo $w+y=f(g(y)+v)$, logo como $g=f^{-1}$ temos 
  \begin{align*}
    g(w+y)=g(f(g(y)+v))=g(y)+v.
  \end{align*}
  Assim $v=g(w+y)-g(y)$.\\
  Como $f$ ĂŠ diferenciĂĄvel em $x$ temos que
  \begin{equation}\label{eq:3.6-eq1}
    w=f(x+v)-f(x)=f^{\prime}(x)\cdot v+r(v).
  \end{equation}
  Assim, $f^{\prime}(x)$ invertivel, implica
  \begin{align*}
    \left[ f^{\prime}(x) \right]^{-1}\cdot w=v+\left[ f^{\prime}(x) \right]^{-1}\cdot r(v).
  \end{align*}
  i.e,
  \begin{align*}
    v=\left[ f^{\prime}(x) \right]^{-1}\cdot w-\left[ f^{\prime}(x) \right]^{-1}\cdot r(v).
  \end{align*}
  EntĂŁo, como queremos provar que $g$ ĂŠ diferenciĂĄvel em $y=f(x)$ e $g^{\prime}(y)=\left[ f^{\prime}(x) \right]^{-1}$, logo
  \begin{align*}
    v=g(w+y)-g(y)=\left[ f^{\prime}(x) \right]^{-1}\cdot w-\left[ f^{\prime}(x) \right]^{-1}\cdot r(v).
  \end{align*}
  \textbf{AfirmaĂ§ĂŁo:} $\lim_{w \to 0}\frac{\left[ f^{\prime}(x) \right]^{-1}\cdot r(v)}{|w|}=0$.\\
  De fato
  \begin{align*}
    \frac{\left|\left[ f^{\prime}(x) \right]^{-1}\cdot r(v)\right|}{|w|}\leq \left| \left[ f^{\prime}(x)\right]^{-1} \right|\cdot \frac{|r(v)|}{|w|}\leq \left| \left[ f^{\prime}(x) \right]^{-1} \right|\cdot \frac{|r(v)|}{|v|}\cdot \frac{|v|}{|w|}.
  \end{align*}
  Logo, pelo teorema \ref{teo:3.6-teorema2}, existe uma vizinhanĂ§a de $x$ onde
  \begin{align*}
    |f(x+v)-f(x)|\geq c|v|
  \end{align*}
  para algĂşm $c>0$, ou seja que usando \eqref{eq:3.6-eq1} temos que $|w|\geq c|v|$, i.e, $\frac{|v|}{|w|}\leq \frac{1}{c}$. Por tanto
  \begin{align*}
    \lim_{w \to 0}\frac{\left| \left[ f^{\prime}(x) \right]^{-1}\cdot r(v) \right|}{|w|}\leq \lim_{w \to 0}\left| \left[ f^{\prime}(x) \right]^{-1} \right|\cdot \frac{|r(v)|}{|v|}\cdot \frac{|v|}{|w|}\leq \frac{M}{c}\lim_{v \to 0}\frac{|r(v)|}{|v|}=0,
  \end{align*}
  Com $M\geq \left|\left[ f^{\prime}(x) \right]^{-1} \right|$, o que conclui o resultado. 
\end{proof}
\begin{corollary}
  Se $f:U\to V$ ĂŠ um homeomorfismo de classe $C^{1}$ e $f^{\prime}(x)$ ĂŠ invertivel para todo $x\in U$, entĂŁo $g=f^{-1}\in C^{1}$. 
\end{corollary}
\begin{proof} 
  Pelo teorema \ref{teo:3.6-teorema-difeo-inverso} $g$ ĂŠ diferenciĂĄvel em $v$ e $g^{\prime}=inv\circ f^{\prime}\circ g$. Logo $g^{\prime}$ ĂŠ contĂ­nua, i.e, $g=f^{-1}\in C^{1}$. 
\end{proof}
\begin{lemma}\label{lema:3.6-lema1}
  Sejam $U\subseteq \mathbb{R}^{m}$ aberto e $g:U\to\mathbb{R}^{n}$ diferenciĂĄvel em $a\in U$ com $g^{\prime}(a)$ sobrejetive. Se $a$ ĂŠ ponto mĂ­nimo local de $|g(x)|$, entĂŁo $g(a)=0$.
\end{lemma}
\begin{proof} 
  Defina $\phi(x):=|g(x)|^{2}=\left\langle 
  g(x),g(x)\right\rangle$ com $x\in U$.\\
  Se $a$ ĂŠ mĂ­nimo local de $|g|$, entĂŁo tambĂŠm ĂŠ mĂ­nimo local de $\phi$.\\
  Logo, $\phi^{\prime}(a)=0$ e como $\phi^{\prime}(x)\cdot v=2\left\langle g^{\prime}(x)\cdot v,g(x) \right\rangle$ para todo $v\in\mathbb{R}^{m}$, temos que 
  $$\phi^{\prime}(a)\cdot v=2\left\langle g^{\prime}(a)\cdot v,g(a) \right\rangle=0\text{ para todo }v\in\mathbb{R}^{m},$$
  logo como $g^{\prime}(a)$ e sobrejetiva, entĂŁo $g(a)=0$. 
\end{proof}
\begin{theorem}[Teorema da aplicaĂ§ĂŁo inversa]
  Seja $f:U\subseteq \mathbb{R}^{m}\to\mathbb{R}^{m}$ de classe $C^{k}$ ($k\geq 1$). Se $a\in U$ e $f^{\prime}(a)$ ĂŠ invertivel, entĂŁo existe uma bola aberta $B=B(a,\delta)\subseteq U$ tal que $F\big|_{B}$ ĂŠ um $C^{k}$-difeomorfismo sobre um aberto $V\subseteq\mathbb{R}^{m}$ contendo $f(a)$. 
\end{theorem}
\begin{proof} 
  \begin{enumerate}
    \item Como $f^{\prime}(a)$ ĂŠ invertivel (em particular injetiva) pelo teorema \ref{teo:3.6-teorema2}, existem $\delta>0$ e $c>0$ taes que $|f(x)-f(y)|\geq c|x-y|$ para todo $x,y\in \overline{B}(a,\delta)$. Note que em particular $f\big|_{\overline{B}}$ ĂŠ injetiva.
    \item $f\big|_{\overline{B}}$ ĂŠ um homeomorfismo sobre $f\left( \overline{B} \right)$. Ja que sabemos que $f\big|_{\overline{B}}$ ĂŠ contĂ­nua, bijetiva e $g=f^{-1}$ ĂŠ contĂ­nua.
    \item $f(B)$ ĂŠ aberto.\\
      \textcolor{red}{Sabemos que $g=f^{-1}$ ĂŠ contĂ­nua e que $f$ ĂŠ bijetiva, logo como $B$ ĂŠ aberto, entĂŁo $f^{-1}\left( B \right)=f(B)$ ĂŠ aberto. (Funciona?)}\\
      Seja $q=f(p)$ com $p\in B$. Seja $S=\partial\overline{B}=\{x\in\mathbb{R}^{m}:|x-a|=\delta\}$, como $f$ ĂŠ injetiva em $\overline{B}$, entĂŁo $q\notin f(S)$. Como $f(S)$ e compacto, logo existe $\epsilon>0$ tal que
      \begin{align*}
        |f(x)-q|\geq 2\epsilon\text{ para todo }x\in S.
      \end{align*}
      \textbf{AfirmaĂ§ĂŁo:} $B(q,\epsilon)\subseteq f(B)$.\\
      Seja $y\in B(q,\epsilon)$ e considere a funĂ§ĂŁo $h(x)=|f(x)-y|$ com $x\in \overline{B}$. Como $\overline{B}$ ĂŠ compacto, $h$ atingue um mĂ­nimo em algum $x_0\in \overline{B}$.
      \begin{itemize}
        \item $x_0\in S$, entĂŁo $|h(x_0)|\geq 2\epsilon$.\\
          Logo $|f(x)-y|\geq |f(x)-q|-|q-y|\geq 2\epsilon-\epsilon\geq \epsilon.$
        \item Por outro lado $h(p)=|q-y|<\epsilon$.\\
          EntĂŁo o mĂ­nimo nĂŁo pode ocorrer em $S$ pois em $p\in B$ o valor ĂŠ menor pelo lema \ref{lema:3.6-lema1} aplicando a $g(x)=f(x)-y$, como $x_0$ ĂŠ ponto mĂ­nimo de $|g(x)|$, temos $g(x_0)=0$, seja $f(x_0)=y$, entĂŁo $y\in f(B)$. 
      \end{itemize}
    \item Assim, pelo teorema \ref{teo:3.6-teorema-difeo-inverso} temos que $f\big|_{B}$ ĂŠ difeomorfismo.
    \item $f\in C^{k}$ ($k\geq 1$), logo $f$ ĂŠ um $C^{k}$-difeomorfismo e pelo teorema \ref{teo:3.6-teorema1}, entĂŁo $g\in C^{k}$.
  \end{enumerate}
\end{proof}
