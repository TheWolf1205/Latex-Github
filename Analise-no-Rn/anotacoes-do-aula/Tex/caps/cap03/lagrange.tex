\begin{definition}[SubmerĂ§ĂŁo]
  Dado um abertoo $U\subseteq\mathbb{R}^{m}$ e uma aplicaĂ§ĂŁo diferenciĂĄvel $f:U\to\mathbb{R}^{m}$ ĂŠ chamada uma \textbf{submerĂ§ĂŁo} quando $f^{\prime}(x)$ ĂŠ sobrejetiva para todo $x\in U$. 
\end{definition}
\textbf{Problema:}\\
Considere $f:U\to\mathbb{R}$ de classe $C^{1}$ definida no aberto $U\subseteq \mathbb{R}^{n+1}$ e $M=\phi^{-1}(c)$, onde $\phi:U\to\mathbb{R}$ ĂŠ $C^{1}$ e $c$ um valor regular de $\phi$ ($\nabla\phi(x)\neq 0$ para todo $x\in M$).\\
Queremos encontrar os "pontos crĂ­ticos" de $f\big|_{M}:M\to\mathbb{R}$.
\begin{definition}[Ponto crĂ­tico]
  Um ponto $x\in M$ ĂŠ um ponto crĂ­tico quando para todo caminho diferenciĂĄvel $\lambda:(-\delta,\delta)\to M$ com $\lambda(0)=x$ tem-se $(f\circ \lambda)^{\prime}(0)=0$. 
\end{definition}
\begin{note}
  Note que se $v=\lambda^{\prime}(0)\in T_{x}M$, isto equivale a $\left\langle \nabla f(x),v \right\rangle=0$ para todo $v\in T_{x}M$, i.ĂŠ, $\nabla f(x)\perp T_{x}M$, \textbf{uma caracterizaĂ§ĂŁo de espaĂ§o tangente}.\\
  Como $M=\phi^{-1}(c)$ e $c$ ĂŠ um valor regular, temos:
  \begin{align*}
    T_{x}M&=\{v\in\mathbb{R}^{n+1}:\left\langle \nabla\phi(x),v \right\rangle=0\}\\
    &=\left[ \nabla\phi(x) \right]^{\perp}.
  \end{align*}
  Alem disso, como $\dim \left[ T_xM \right]^{\perp}=1$, temos que $\nabla f(x)\perp T_{x}M$ se, somente se, $\nabla f(x)=\lambda\nabla \phi(x)$ para todo $\lambda\in\mathbb{R}$. 
\end{note}
\begin{theorem}[Multiplicadores de Lagrange (com uma restriĂ§ĂŁo)]
  O ponto $x\in U$ ĂŠ ponto crĂ­tico da restriĂ§ĂŁo $f\big|_{M}$ onde $M=\phi^{-1}(c)$ se, somente se
  \begin{enumerate}
    \item $\phi(x)=c$.
    \item $\nabla f(x)=\lambda\nabla(x)$ para todo $\lambda\in\mathbb{R}$. 
  \end{enumerate}
  \textbf{Caso geral}\\
  Considere
  \begin{itemize}
    \item $f:U\to\mathbb{R}$ de classe $C^{1}$, $U\subseteq \mathbb{R}^{n+m}$.
    \item $g=(g_1,\cdots,g_m):U\to\mathbb{R}^{m}$ $C^{1}$ uma submerĂ§ĂŁo, i.ĂŠ, $g^{\prime}(p)$ ĂŠ sobrejetiva para todo $p\in M$.
    \item $M=g^{-1}(c)$, com $c\in\mathbb{R}^{m}$ valor regular. 
  \end{itemize}
  Se $p\in M$ ĂŠ um extremo local de $f\big|_{M}$, entĂŁo existem $\lambda_{1},\cdot,\lambda_{m}\in\mathbb{R}$ taes que
  \begin{align*}
    \nabla f(p)=\sum_{i=1}^{m}\lambda_{i}\nabla g_{i}(p).
  \end{align*}
\end{theorem}
\begin{proof} 
  Seja $p=(x_0,y_0)\in M$ um extremo local de $f\big|_{M}$. Como $g$ ĂŠ uma submerĂ§ĂŁo, $g^{\prime}(p)$ ĂŠ sobrejetiva.
  \begin{align*}
    g^{\prime}:&U\to L(\mathbb{R}^{n+m},\mathbb{R}^{m})\\
    &Jg(p)\in M_{m\times (n+m)}.
  \end{align*}
  Sem perdida de generalidade suponha
  \begin{align*}
    Jg(p)=(x_1,\cdots,x_n,\underbrace{y_1,\cdots,y_m}_{\text{columas L.I}}).
  \end{align*}
  Logo $g^{\prime}(p)\big|_{\{0\}\times \mathbb{R}^{m}}$ ĂŠ um isomorfismo sobre $\mathbb{R}^{m}$.\\
  Pelo teorema da funĂ§ĂŁo implicita, existem um aberto $A\subseteq \mathbb{R}^{n}$ com $x\in A$ e uma aplicaĂ§ĂŁo $\xi:A\to\mathbb{R}^{m}$ de classe $C^{1}$ taes que $g(x,\xi(x))=c$ e $\xi(x_0)=y_0$. Alem disso, o grĂĄfico de $\xi$ estĂĄ contido em $M$.\\
  Definimos o subespaĂ§o vetorial $G=\{(h,\xi^{\prime}
  (x_0)\cdot h):h\in\mathbb{R}^{n}\}$.\\
  Note que $\{(e_1,\xi^{\prime}(x_0)\cdot e_1),\cdots,(e_n,\xi^{\prime}(x_0)\cdot e_n)\}$ ĂŠ uma base de $G$, entĂŁo $\dim G=m$.\\
  \textbf{AfirmaĂ§ĂŁo:} $T_{p}M=G$.\\
    Como $g(x,\xi(x))=c$, entĂŁo $g^{\prime}(p)\cdot(h,\xi^{\prime}(x_0)\cdot h)=0$, logo $(h,\xi^{\prime}(x_0)\cdot h)\in \Ker g^{\prime}(p)=T_{p}M$, assim $G\subseteq T_{p}M$, logo como tem a mesma dimenĂ§ĂŁo, entĂŁo $G=T_{p}M$.\\
  Agora, para cada $v=(h,\xi^{\prime}(x_0)\cdot h)\in G$ temos que
  \begin{align*}
    0=g^{\prime}_{j}(p)\cdot v=\left\langle \nabla g_{j}(p),v \right\rangle,\quad\text{ para todo }j=1,\cdots,m.
  \end{align*}
  EntĂŁo $\nabla g_{j}(p)\in G^{\perp}$ para todo $j$.\\
  AlĂŠm disso como $\dim G + \dim G^{\perp}=n+m$, entĂŁo $\dim G^{\perp}=m$.\\
  Portanto, note que como $B=\{\nabla g_1(p),\cdots,\nabla g_m(p)\}$ ĂŠ L.I, entĂŁo $B$ ĂŠ base de $G^{\perp}$.\\
  \textbf{AfirmaĂ§ĂŁo:} $\nabla f(p)\in G^{\perp}$.\\
    Como $p$ ĂŠ extremo local de $f\big|_{M},x_{0}$ ĂŠ extremo local de $\eta(x)=f(x,\xi(x))$. EntĂŁo para todo $h\in\mathbb{R}^{n}$, temos que
    \begin{align*}
      0=f^{\prime}(x_0,\xi(x_0))\cdot(h,\xi^{\prime}(x_0)\cdot h)=\left\langle \nabla f(p),(h,\xi^{\prime}(x_0)\cdot h) \right\rangle.
    \end{align*}
    Logo $\nabla f(p)\in G^{\perp}$.\\
  Assim, existe $\lambda_1\cdots,\lambda_{m}\in\mathbb{R}$ taes que 
  \begin{align*}
    \nabla f(p)=\sum_{i=1}^{m}\lambda_{i}\nabla g_{i}(p).
  \end{align*}
\end{proof}
\begin{example}
  Seja $f(x,y)=ax+by$ com $a^{2}+b^{2}\neq 0$.\\
  Determine os pontos crĂ­ticos de $f\big|_{S^{1}}$, onde
  \begin{align*}
    S_1=\{(x,y)\in\mathbb{R}^{2}:x^{2}+y^{2}=1\}.
  \end{align*}
  Note que a restriĂ§ĂŁo $x^{2}+y^{2}=1$ se pode escrever como $\phi(x)=x^{2}+y^{2}-1$, assim temos
  \begin{itemize}
    \item $\nabla f(x,y)=(a,b)$.
    \item $\nabla g(x,y)=(2x,2y)$.
  \end{itemize}
  Pelo teorema de multiplicadores de Lagrange, os pontos crĂ­ticos satisfazen
  \begin{align*}
    \begin{cases}
      \nabla f(x,y)=\lambda \nabla g(x,y), \\
      g(x,y)=0.
    \end{cases}
  \end{align*}
  Assim, $a=2\lambda x$ e $b=2\lambda y$, entĂŁo $\displaystyle \frac{a^{2}}{4\lambda^{2}}+\frac{b^{2}}{4\lambda^{2}}=1$, portanto $\lambda=\pm\frac{\sqrt{a^{2}+b^{2}}}{2}$.\\
  Logo
  \begin{align*}
    \begin{cases}
      f\left( \frac{a}{2\lambda},\frac{b}{2\lambda} \right)=\frac{a^{2}+b^{2}}{2\lambda}=\sqrt{a^{2}+b^{2}}, &\text{ ĂŠ o mĂĄximo } \lambda>0 \text{,} \\
      f\left( \frac{a}{2\lambda},\frac{b}{2\lambda} \right)=\frac{a^{2}+b^{2}}{2\lambda}=-\sqrt{a^{2}+b^{2}}, &\text{ ĂŠ o mĂ­nimo } \lambda<0.
    \end{cases} 
  \end{align*}
\end{example}

\begin{example}[Desigualdade de Hadamard]
  Se $x\in M_{n\times n}$ com linhas $x_1,\cdots,x_n$, entĂŁo
  \begin{align*}
    |\det(x)|\leq |x_1|\cdots|x_n|.
  \end{align*}
  \textbf{EsboĂ§oda prova:}\\
  Se $\det(x)=0$, a desigualdade ĂŠ trivial. Caso contrario, escrevemos $x_i=|x_i|w_i$ com $|w_i|=1$, logo escrevemos $x=DW$ com $D$ definida por 
  \begin{align*}
    x=\begin{pmatrix}
      |x_1| & 0 & 0\\
      0 & \ddots & 0\\
      0 & 0 & |x_n|
    \end{pmatrix}W.
  \end{align*}
  Assim $\det(x)=\det(D)\det(W)$, portanto provando que $\det(W)\leq 1$ temos que
  \begin{align*}
    |\det(x)|\leq |x_1|\cdots|x_n|.
  \end{align*}
\end{example}
