\begin{definition}[Funcão uniformemente diferenciável]
  Seja $f:U\subseteq \mathbb{R}^{m}\to\mathbb{R}^{n}$ de classe $C^{1}$ no aberto $U$. Se para todo $\epsilon>0$ existe $\delta>0$ ($\delta$ não depende de $x$) tal que se para todo $v\in \mathbb{R}^{m}$ com $|v|< \delta$ ten-se que $|f(x+v)-f(x)-f^{\prime}(x)\cdot v|< \epsilon|v|$ para todo $x\in U$, diremos que $f$ é uma funcão \textbf{uniformemente diferenciável em $U$}. 
\end{definition}
\begin{proposition}[Teorema Fundamental do Cálculo para caminhos]
  Se $f:[a,b]\to\mathbb{R}^{n}$ é um caminho diferenciável de classe $C^{1}$, então
  \begin{align*}
    \int_{a}^{b}f^{\prime}(t)\, dt=f(b)-f(a).
  \end{align*}
\end{proposition}
\begin{theorem}
  Seja $f:U\subseteq \mathbb{R}^{m}\to\mathbb{R}^{n}$ de classe $C^{1}$ no aberto $U$. Se $K\subseteq U$ é compacto, então $f$ é uniformemente diferenciável em $K$. 
\end{theorem}
\begin{proof} 
  Sabemos que existe $\delta>0$ tal que $B(x,2\delta)\subseteq U$ para todo $x\in K$. (\textcolor{red}{Por que?})

  Defina $L=\bigcup_{k\in K}B[x,\delta]=\{y\in \mathbb{R}^{n}:d(y,k)\leq \delta,k\in K\}$.\\
  \textbf{Afirmação:} $L$ é compacto.

  Veamos que $L$ é limitado. Note que como $K$ é compacto, então existe $M>0$ tal que para todo $x\in K$ esta cumprido que $|x|< M$, logo temos que $|y|\leq |y-x|+|x|\leq \delta + M$ i,e. $L$ é limitado.

  Veamos que $L$ é fechado. Seja $\{y_n\}_{n\in\mathbb{Z}^{+}}\subseteq L$ com $y_n\to y$, vamos ver que $y\in L$. Note que para cada $n\in\mathbb{Z}^{+}$, temos que existe $x_n\in K$ tal que $|x_n-y_n|\leq \delta$, logo como $K$ é compacto, $\{x_n\}_{n\in\mathbb{Z}^{+}}$ possui uma subsequencia $x_{n_k}\to x$ com $x\in K$, assim
  \begin{align*}
    \lim_{n_k \to \infty}|y-x|&\leq \lim_{n_k \to \infty}|y-y_{n_k}|+|y_{n_k}-x_{n_k}|+|x_{n_k}-x|\\
    &\leq 0 + \delta + 0.
  \end{align*}
  Então $y\in B[x,\delta]\subseteq L$.\\
  Assim, $L$ é compacto.

  Logo se $x\in K$ e $|v|\leq \delta$, então $x+v\in L$.

  Note que $f^{\prime}:L\to L(\mathbb{R}^{m},\mathbb{R}^{n})$ é uniformemente contínua.

  Logo, podemos asumir que $|f^{\prime}(x+v)-f^{\prime}(x)|<\epsilon$ para toddo $x\in K$ e para todo vetor $v\in\mathbb{R}^{m}$ com $|v|<\delta$.

  Assim, para todo $t\in[0,1]$, temos $|tv|\leq t|v|\leq |v|\leq \delta$ e assim $|f^{\prime}(x+tv)-f^{\prime}(x)|<\epsilon$ para todo $t\in[0,1]$.

  Considere $\lambda:[0,1]\to\mathbb{R}^{n}$ dada por $\lambda(t)=f(x+tv)$, logo como $f$ é $C^{1}$, $\lambda$ é $C^{1}$, então pelo teorema fundamental do cálculo para caminhos temos que
  \begin{align*}
    f(x+v)-f(x)=\lambda(1)-\lambda(0)=\int_{0}^{1}\lambda^{\prime}(t)\,dt=\int_{0}^{1}f^{\prime}(x+tv)\cdot v\,dt.
  \end{align*}
  Então
  \begin{align*}
    |f(x+v)-f(x)-f^{\prime}(x)\cdot v|&=\left| \int_{0}^{1}f^{\prime}(x+tv)\cdot v\,dt -f^{\prime}(x)\cdot v\right|\\
    &\leq \int_{0}^{1}\left| f^{\prime}(x+tv)-f^{\prime}(x) \right||v|\,dt\\
    &\leq \epsilon |v||[0,1]|\\
    &\leq \epsilon |v|.
  \end{align*}
\end{proof}
\begin{lemma}
  Se $f:I\to\mathbb{R}^{n}$ diferenciável no ponto $c\in I$. Dadas as sequencias de números $a_k\neq b_k$ em $I$, com $a_k\leq c \leq b_k$ e $\lim a_k=\lim b_k=c$, tem-se
  \begin{align*}
    f^{\prime}(c)=\lim_{k \to \infty}\frac{f(b_k)-f(a_k)}{b_k-a_k}.
  \end{align*}
\end{lemma}
\begin{lemma}\label{lem:2-formula-de-taylor}
  Sejam $f:[a,b]\to\mathbb{R}^{n}$ e $\varphi:[a,b]\to\mathbb{R}^{n}$ contínuas e diferenciávels no intervalo $(a,b)$, se $|f^{\prime}(t)|\leq \varphi^{\prime}(t)$ com $\varphi^{\prime}(t)>0$ para todo $t\in(a,b)$. Então $|f(b)-f(a)|\leq \varphi(b)-\varphi(a)$. 
\end{lemma}
\begin{theorem}[Fórmula de Taylor para caminhos]
  Seja $f:[a,b]\to\mathbb{R}^{n}$ um caminho de classe $C^{p-1}$ $p$ vezes diferenciável em $(a,a+h)$. Se $|f^{(p)}(t)|\leq M$ para todo $t\in(a,a+h)$, então
  \begin{align*}
    f(a+h)=f(a)+hf^{\prime}(a)+\cdots+\frac{h^{p-1}}{(p-1)!}f^{(p-1)}(a)+r_p,\quad\text{ onde }|r_p|\leq\frac{Mh^{p}}{p!}.
  \end{align*}
  Colocando $a+h=b$, obtemos
  \begin{align*}
    f(b)=f(a)+(b-a)f^{\prime}(a)+\cdots+\frac{(b-a)}{(p-1)!}f^{(p-1)}(a)+r_p,\quad\text{ onde }|r_p|\leq\frac{M(b-a)^{p}}{p!}.
  \end{align*}
\end{theorem}
\begin{proof} 
  Defina o caminho $g:[a,b]\to\mathbb{R}^{n}$ por
  \begin{align*}
    g(t)=f(t)+(b-t)f^{\prime}(t)+\cdots+\frac{(b-t)}{(p-1)!}f^{(p-1)}(t).
  \end{align*}
  Então, $g$ é contínua em $[a,b]$ e diferenciável em $(a,b)$ com
  \begin{align*}
    g^{\prime}(t)=\frac{(b-t)}{(p-1)!}f^{(p)}(t).
  \end{align*}
  Logo $|g^{\prime}(t)|\leq \frac{M(b-t)^{p-1}}{(p-1)!}$.\\
  Seja $\phi(t)=\frac{M(b-t)^{p}}{p!}$, então pelo lema \ref{lem:2-formula-de-taylor}, temos
  \begin{align*}
    |r_p|&=\left|f(b)-f(a)-(b-a)f^{\prime}(a)-\cdots-\frac{(b-a)}{(p-1)!}f^{(p-1)}(a)\right|\\
    &=|g(b)-g(a)|\\
    &\leq \varphi(b)-\varphi(a)\\
    &=\frac{M(b-a)^{p}}{p!}.
  \end{align*}
\end{proof}
no contexto de funções, temos o seguinte.

A fórmula de Taylor para uma função $f:U\to\mathbb{R}$ diferenciável no aberto $U\subseteq \mathbb{R}^{n}$ é dada por
\begin{align*}
  f(a+v)=f(a)+df(a)\cdot v+\frac{1}{2}d^{2}f(a)\cdot v^{2}+\cdots+\frac{1}{p!}d^{p}f(a)\cdot v^{p}+r_p(v).
\end{align*}
\begin{note}
  Lembre que
  \begin{itemize}
    \item 
      \begin{align*}
        df:U&\to L(\mathbb{R}^{n},\mathbb{R})\\
        a&\to df(a):\mathbb{R}^{n}\to\mathbb{R}^{n},\quad\text{linear.}
      \end{align*}
      \begin{align*}
        df(a)\cdot v=\left\langle \nabla f(a),v \right\rangle=\sum_{i=1}^{n}\frac{\partial f}{\partial x_i}(a)\alpha_i,\quad\text{onde }v=(\alpha_1,\cdots,\alpha_n).
      \end{align*}
    \item $d^{2}f(a)\in L_2(\mathbb{R}^{n},\mathbb{R})=L(\mathbb{R}^{n},L(\mathbb{R}^{n},\mathbb{R}))$. Na prática, $d^{2}f(a)$ é representa pela matriz Hessiana $Hf(a)$ (simétrica se $f\in C^{2}$). 
      \begin{align*}
        d^{2}f(a)\cdot v^{2}=\sum_{i,j=1}^{n}\frac{\partial f}{\partial x_i\partial x_j}\cdot\alpha_i\alpha_j=v^{T}Hf(a)v.
      \end{align*}
    \item $d^{p}f(a)\in L_p(\mathbb{R}^{n},\mathbb{R})$ é o espaço das aplicações $p$-lineares de $\left( \mathbb{R}^{n} \right)^{p}$ em $\mathbb{R}$. Na prátic
      \begin{align*}
        d^{p}f(a)\cdot v^{p}=\sum_{i_1,\cdots,i_p=1}^{n}\frac{\partial f}{\partial x_{i_1}\cdots\partial x_{i_p}}(a)\alpha_{i_1}\cdots\alpha_{i_p}.
      \end{align*}
  \end{itemize}
\end{note}
\begin{note}[Resto]
  \phantom{x}
  \begin{enumerate}
    \item Fórmula de Taylor infinitesimal.

      Se $f$ é $p$-vezes diferenciável no ponto $a$, então
      \begin{align*}
        \lim_{v\to 0}\frac{r_p(v)}{|v|}=0.
      \end{align*}
    \item Resto de Lagrange.

      Assumindo $[a,a+v]\subseteq U,f\in C^{p}$, $p+1$ vezes diferenciável no segmento aberto $(a,a+v)$, então existe $\theta\in(0,1)$ tal que 
      \begin{align*}
        r_p(v)=\frac{1}{(p+1)!}d^{p+1}f(a+\theta v)\cdot v^{p+1}\xrightarrow[v\to 0]{}0.
      \end{align*}
    \item Resto integral.

      Se $f\in C^{p+1}$ e $[a,a+v]\subseteq U$, então
      \begin{align*}
        r_p(v)=\frac{1}{p!}\int_{0}^{1}f(a+\theta v)\cdot v^{p+1}\,dt.
      \end{align*}
  \end{enumerate}
\end{note}
\begin{theorem}[Fórmula de Taylor com resto de Lagrange para funções reais]
  Seja $f:U\subseteq\mathbb{R}^{m}\to\mathbb{R}$ de classe $C^{p}$, $p+1$ vezes diferenciável no segmento aberto $(a,a+v)\subseteq U$. Se $|f^{(p)}(t)|\leq M$ para todo $t\in (a,a+h)$, então
  \begin{align*}
    f(a+v)=f(a)+df(a)\cdot v+\frac{1}{2}d^{2}f(a)\cdot v^{2}+\cdots+\frac{1}{p!}d^{p}f(a)\cdot v^{p}+r_p(v).
  \end{align*}
\end{theorem}
\begin{proof} 
  Fixemos $a\in U$ e $v\in\mathbb{R}^{n}$ tal que $a+tv\in U$ para todo $t\in[0,1]$.\\
  Defina $\varphi(t)=f(a+tv)$, com $t\in[0,1]$.\\
  Note que
  \begin{align*}
    \varphi^{\prime}(t)&=df(a+tv)\cdot v\\
    \varphi^{\prime\prime}(t)&=d^{2}f(a+tv)\cdot v^{2}\\
    \varphi^{(p)}(t)&=d^{p}f(a+tv)\cdot v^{p}.
  \end{align*}
  A função $\varphi$ é $p+1$ vezes diferenciável em $(0,1)$ com $\varphi^{(p)}$ contínua em $[0,1]$.\\
  Pelo teorema de Taylor para funções de uma variável existe $\theta\in[0,1]$ tal que
  \begin{align*}
    \varphi(1)=\varphi(0)+\varphi^{\prime}(0)+\frac{1}{2}\varphi^{\prime\prime}(0)+\cdots+r_p(\theta,v).
  \end{align*}
  Sabemos que $\varphi(1)=f(a+v),\varphi(0)=f(a)$ e $\varphi^{(j)}(0)=d^{j}f(a)\cdot v^{j}$ para todo $j=1,\cdots,p$. Sustituindo
  \begin{align*}
    f(a+v)=f(a)+df(a)\cdot v+\cdots+\frac{1}{p!}d^{p}f(a)\cdots v^{p}+\underbrace{\frac{1}{(p+1)!}d^{p+1}f(a+\theta v)\cdot \theta v^{p+1}}_{r_p(v)}.
  \end{align*}
  Exercicio!
\end{proof}
O resto éuma função $r_p:U_0\to\mathbb{R}$ definida no aberto $U_0=\{v\in\mathbb{R}^{n}:a+v\in U\}$ por
\begin{align*}
  r_p(v)=f(a+v)-f(a)-df(a)\cdot v-\cdots-\frac{1}{p!}d^{p}f(a)\cdot v^{p}.
\end{align*}
Note que $r_p$ é $p$-vezes diferenciável no ponto $0$ e $r_p^{\prime}(0)=0$.
\begin{lemma}
  Seja $r:U\to\mathbb{R}$ uma função $p$-vezes diferenciável no ponto $0\in U$. Se
  \begin{align*}
    r(0)=\frac{\partial r}{\partial x_i}(0)=\frac{\partial r}{\partial x_i\partial x_j}(0)=\cdots=\frac{\partial r}{\partial x_{i_1}\cdots\partial x_{i_p}}(0)=0.
  \end{align*}
  Então $\displaystyle \lim_{v \to 0}\frac{r(v)}{|v|}=0$.
\end{lemma}
\begin{proof} 
  Usaremos indução em $p$.\\
  $p=1$.\\
  Como $r$ é diferenciável em $0$ ten-se que
  \begin{align*}
    r(v)=r(0)+dr(0)\cdot v+s(v)=s(v),\quad\text{onde }\lim_{v \to 0}\frac{s(v)}{|v|}=0.
  \end{align*}
  Então
  \begin{align*}
    \lim_{v \to 0}\frac{r(v)}{|v|}=0.
  \end{align*}
  Paso indutivo.\\
  Suponha que para funções que cumprem $r(0)=d^{k}r(0)=0$ para $k=1,\cdots,p$, temos 
  \begin{align*}
    \lim_{v \to 0}\frac{r(v)}{|v|^{p}}=0
  \end{align*}
  Seja $r$ $p+1$-vezes diferenciável com $d^{p+1}r(0)=0$.\\
  Fixemos $v\neq 0$. Pelo teorema de valor médio de uma variavél real aplicado a $\phi(t)=r(tv)$ com $t\in[0,1]$, existe $\theta\in(0,1)$ tal que $\phi(1)-\phi(0)=\phi^{\prime}(\theta)$. Como $\phi(0)=r(0)=0$, temos
  \begin{align*}
    r(v)=\phi(1)=\phi^{\prime}(\theta)=dr(\theta v)\cdot v=\sum_{i=1}^{n}\frac{\partial r}{\partial x_i}(\theta v)\cdot \alpha_i,\quad\text{onde }v=(\alpha_1,\cdots,\alpha_n).
  \end{align*}
  Logo
  \begin{align*}
    \frac{r(v)}{|v|^{p+1}}=\sum_{i=1}^{n}\frac{\frac{\partial r}{\partial x_i}(\theta v)}{|v|^{p}}\cdot \frac{\alpha_i}{|v|}.
  \end{align*}
  Usando la hipótesis indutiva e fazendo $v\to 0$ temos
  \begin{align*}
    \lim_{v \to 0}\frac{r(v)}{|v^{p+1}|}=0.
  \end{align*}
\end{proof}

Agora, no contexto de aplicações, temos o seguinte.\\
Seja $f:U\subseteq\mathbb{R}^{m}\to\mathbb{R}^{n}$ uma aplicação diferenciável no aberto $U$ com $a,a+v\in U$. A fórmula de Taylor se escreve
\begin{equation}\label{eq:formula-de-taylor-para-aplicacoes}
  f(a+v)=f(a)+f^{\prime}(a)\cdot v+f^{\prime\prime}(a)\cdot v^{2}+\cdots+\frac{1}{p!}f^{(p)}(a)\cdot v^{p}+r_p(v).
\end{equation}
Onde
\begin{align*}
  f^{(j)}f(a)\cdot v^{j}=\frac{\partial ^{j}f}{\partial v^{j}}(a)=\underbrace{\frac{\partial }{\partial v}\left(\frac{\partial }{\partial v}\left(\cdots\left(\frac{\partial f}{\partial v}\right)\cdots\right)\right)}_{j\text{ vezes}}(a).
\end{align*}
\begin{note}[Resto]
  \phantom{x}
  \begin{enumerate}
    \item Fórmula de Taylor infinitesimal.

      Se $f$ é $p$-vezes diferenciável no ponto $a$, então
      \begin{align*}
        \lim_{v\to 0}\frac{r_p(v)}{|v|}=0.
      \end{align*}
    \item Fórmula de Taylor com resto de Lagrange.

      Sejam $[a,a+v]\subseteq U,f\in C^{p}$, $p+1$ vezes diferenciável em cada ponto do segmento $(a,a+v)$ com $|f^{(p+1)}(x)\cdot w^{p+1}|\leq M|w|^{p+1}$ para todo $x\in (a,a+v)$ e todo $w\in\mathbb{R}^{m}$, então 
      \begin{align*}
        |r_p(v)|\leq \frac{M}{(p+1)!}|v|^{p+1}.
      \end{align*}
    \item Fórmula de Taylor com resto integral.

      Se $f\in C^{p+1}$ e $[a,a+v]\subseteq U$, então
      \begin{align*}
        r_p(v)=\frac{1}{p!}\int_{0}^{1}(1-t)^{p}f^{(p+1)}(a+\theta v)\cdot v^{p+1}\,dt,
      \end{align*}
      onde, para cada $j=1,\cdots,p$, $f^{(j)(a)\cdot v^{j}}$ é o vetor de $\mathbb{R}^{n}$ cujas coordenadas são os números $d^{(j)}f_i(a)\cdot v^{j}$ onde $f=(f_1,\cdots,f_n)$. 
  \end{enumerate}
\end{note}
Note que a fórmula de Taylor \eqref{eq:formula-de-taylor-para-aplicacoes} equivale a $n$ igualdades da fórmula de Taylor para funções reais ($f:U\subseteq \mathbb{R}^{m}\to\mathbb{R}$), pelo que proba $1.$ e $2.$ está pronto.
\begin{theorem}[Fórmula de Taylor com resto de Lagrange para aplicações]
  Seja $f:U\subseteq \mathbb{R}^{m}\to\mathbb{R}^{n}$ uma aplicação no $C^{p}$ e $p+1$ vezes diferenciável em cada ponto do segmento $(a,a+v)\subseteq U$ com $f^{(p+1)}(x)\cdot w^{p+1}$ para todo $x\in (a,a+v)$e todo $w\in\mathbb{R}^{m}$, então 
  \begin{align*}
    f(a+v)=f(a)+f^{\prime}(a)\cdot v+\frac{1}{2}f^{\prime\prime}(a)\cdot v^{2}+\cdots+\frac{1}{p!}f^{(p)}(a)\cdot v^{p}+r_p(v),
  \end{align*}
  onde
  \begin{align*}
    |r_p(v)|\leq \frac{M}{(p+1)!}|v|^{p+1}. 
  \end{align*}
\end{theorem}
\begin{proof} 
  Considere o caminho $\varphi:[0,1]\to\mathbb{R}^{n}$ dado por $\varphi(t)=f(a+tv)$.\\
  Sabemos que
  \begin{itemize}
    \item $\varphi\in C^{p}$.
    \item $\varphi$ é $p+1$ vezes diferenciável em $(0,1)$.
    \item $\varphi(0)=f(a)$, $\varphi(1)=f(a+v)$.
    \item $\varphi^{(j)}(t)=d^{j}f(a+tv)\cdot v^{j}$ para $j=1,\cdots,p,p+1$.
    \item $|\varphi^{(p+1)}(t)|=|d^{p+1}f(a+tv)\cdot v^{p+1}|\leq M|v|^{p+1}$ para todo $t\in(0,1)$. 
  \end{itemize}
  Logo, pela fórmula de Taylor para caminhos
  \begin{align*}
    \varphi(1)=\varphi(0)+\varphi^{\prime}(0)+\frac{1}{2}\varphi^{\prime\prime}(0)+\cdots+\frac{1}{p!}\varphi^{(p)}(0)+r_p,\quad\text{onde }|r_p|\leq\frac{M|v^{p+1}|}{(p+1)!}.
  \end{align*}
  i,e.
  \begin{align*}
    f(a+v)=f(a)+df(a)\cdot v+\cdots+\frac{1}{p!}d^{(p)}f(a)\cdot v^{p}+r_p(v),\quad\text{onde }|r_p(v)|\leq \frac{M|v|^{p+1}}{(p+1)!}.
  \end{align*}
\end{proof}
