\begin{proposition}
  Seja $M$ uma superficie. EntĂŁo, para cada $k$ existe um Ăşnico operador linear
  \begin{align*}
    d:\Lambda^{k}(M)&\to\Lambda^{k+1}(M),
  \end{align*}

  \begin{enumerate}[i)]
    \item Se $f$ Ă© uma $0$-forma, $df$ Ă© a diferencial de $f$.

      \begin{align*}
        dw=\sum_{J}dw_J\wedge dx^{J}=\sum_{J}\sum_{i=1}^{n}\frac{\partial w_J}{\partial x^{i}}dx^{i}\wedge dx^{J}.

  \end{enumerate}
\end{proposition}

  Dada uma parametrizaĂ§ĂŁo $(\varphi, V_0)$ para $M$, definimos

  \end{align*}
  \textbf{AfirmaĂ§ĂŁo:}\\
  O operador $d$ de $M$ estĂˇ bem definido.\\
  De fato, seja $(\psi, U_0)$ uma outra parametrizaĂ§ĂŁo para vizinhanĂ§a de $V$ de $M$ com $\varphi(U)\cap\psi(U_0)\neq \emptyset$. EntĂŁo,
  \begin{align*}

    =&(\psi^{-1})^{\ast}(\varphi^{-1}\circ\psi)^{\ast}d\left[ \varphi^{\ast}w \right]\\
    =&(\psi^{-1})^{\ast}d\left[ (\varphi^{-1}\circ\psi)^{\ast}\varphi^{\ast}w \right]\\
    =&(\psi^{-1})^{\ast}d\left[ (\varphi\circ\varphi^{-1}\circ\psi)^{\ast}w \right]\\
    =&(\psi^{-1})^{\ast}d\left[ \psi^{\ast}w \right].
  \end{align*}
  \textbf{Unicidade:}\\
  \textbf{AfirmaĂ§ĂŁo:} Suponha que $w_1$ e $w_2$ sĂŁo $k$-formas tais que $w_1=w_2$ em um aberto $U$, entĂŁo $dw_1=dw_2$ em $U$.\\
  De fato, considete $\eta=w_1-w_2$.\\
  Dado $p\in U$, seja $f$ uma funĂ§ĂŁo lombada suportada em $U$ (como no exemplo \ref{ex:bumb-function}) 
  \begin{align*}

      1, &\text{ se } x\in U_p\subseteq U \text{,} \\
      0, &\text{ se } x\notin U.
    \end{cases}
  \end{align*}

  EntĂŁo temos que $f\eta\equiv 0$, depois temos que
  \begin{align*}
    0=d(f\wedge\eta)=df\wedge\eta+f\wedge d\eta.
  \end{align*}
  Isto Ă©, no ponto $p$, que como $f$ Ă© constante no $U_p$ e $f(p)=1$ temos que 
  \begin{align*}
    0=df_{p}\wedge\eta_{p}+f(p)\wedge d\eta_{p}=d\eta_p.
  \end{align*}
  Para todo $p\in U$, logo temos que $d\eta=0$ em $U$.\\
  Agora, dado $w\in\Lambda^{k}(M)$, escrevala em coordenadas locais em relaĂ§ĂŁo a uma carta $(\varphi,U)$ como $w=\sum_{J}w_Jdx^{J}$.\\

  \end{align*}
  Segue-se que
  \begin{align*}
    dw_p=\sum_{J}(dw_J)_{p}\wedge dx^{J}_p.
  \end{align*}
\end{proof}
\begin{proposition}
  Seja $F:M\to N$ uma aplicaĂ§ĂŁo diferenciĂˇvel entĂŁo
  \begin{align*}

  \end{align*}
\end{proposition}
\begin{proof} 
  Seja $(\varphi,U)$ uma carta para $M$ e $(\psi,V)$ uma carta para $N$, temos
  \begin{align*}
    F^{\ast}(dw)=&F^{\ast}\left( (\psi^{-1})^{\ast}d\left[ \psi^{\ast}w \right] \right)\\
    =&(\psi^{-1}\circ F)^{\ast}d\left[ \psi^{\ast}w \right]\\
    =&(\psi^{-1}\circ F\circ\varphi\circ\varphi^{-1})^{\ast}d\left[ \psi^{\ast}w \right]\\
    =&(\varphi^{-1})^{\ast}(\psi^{-1}\circ F\circ\varphi)^{\ast}d\left[ \psi^{\ast}w \right]\\
    =&(\varphi^{-1})^{\ast}d\left[ (F\circ\varphi)^{\ast}w \right]\\
    =&d\left[ F^{\ast}w \right].
  \end{align*}
\end{proof}
