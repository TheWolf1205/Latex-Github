\begin{proposition}
  Vale
  \begin{align*}
    e^{I}\wedge e^{J}=e^{IJ}.
  \end{align*}
  Em particular,
  \begin{align*}
    e^{I}=e^{i_1}\wedge\cdots\wedge e^{i_k},
  \end{align*}
  de modo que
  \begin{align*}
    e^{i_1}\wedge\cdots\wedge e^{i_k}(e_{j_1},\cdots,e_{j_l})=e^{I}(e_J)=\det\left[ e^{i_r}(e^{j_s}) \right],
  \end{align*}
  Ă© a base exterior
  \begin{align*}
    B^{k}=\{e^{i_1}\wedge\cdots\wedge e^{i_k}:i_{r}\in\{1,\cdots,n\}\, r\in\{1,\cdots,k\}\text{ e $I$ crescente }\}
  \end{align*}
  Ă© base para $\Lambda^{k}(V)$.
\end{proposition}
\begin{proof} 
  Sejam
  \begin{align*}
    I=&i_1,\cdots,i_k.\\
    J=&j_1,\cdots,j_l.\\
    N=&n_1,\cdots,n_{k+l}.\\
    N^{k}=&n_1,\cdots,n_k.\\
    N^{l}=&n_{k+1},\cdots,n_{k+l}.
  \end{align*}
  Note que por multilinearidade temos que basta provar a identidade para os vetores $e^{N}$ da base $\Lambda^{{k+l}}(V)$, isto Ă© vamos ver que $e^{I}\wedge e^{J}(e_N)=e^{IJ}(e_N)$.\\
  De fato temos que
  \begin{itemize}
    \item 
      \begin{align*}
        e^{I}\wedge e^{J}(e_N)=\frac{1}{k!l!}\sum_{\sigma\in S_{k+l}}(\sgn\sigma)e^{I}(e_{N^{k}_{\sigma}})e^{J}(e^{l}_{\sigma}).
      \end{align*}
    \item
      \begin{align*}
        e^{IJ}(e_{N})=\delta^{IJ}_{N}.
      \end{align*}
  \end{itemize}
  Note que se $N$ possui algĂşm Ă­ndice que nĂŁo pertenece nem $I$, nem $J$ ou se $N$ possui Ă­ndices repetidos, entĂŁo ambos lados sĂŁo nulos.\\
  Logo, podemos entĂŁo nos restringir ao caso em que $N$Ă© uma permutaĂ§ĂŁo de $IJ$.\\
  Segue-se que
  \begin{itemize}
    \item $e^{I}(e_{N^{k}_{\sigma}})=\delta^{I}_{N^{k}_{\sigma}}$.
    \item $e^{J}(e_{N^{l}_{\sigma}})=\delta^{J}_{N^{l}_{\sigma}}$.
  \end{itemize}
  De modo que a Ăşnica maneira de obter um valor nĂŁo nulo Ă© se $\sigma$ for uma permutaĂ§ĂŁo que permuta os primeiros $k$ Ă­ndices e os Ăşltimos $l$ Ă­ndices separadamente, logo temos que existe $\tau\in S_{k}$ e $\eta\in S_{l}$ tal que
  \begin{align*}
    \sigma=\tau\eta,\quad\text{e}\quad \sgn\sigma=\sgn\tau\sgn\eta.
  \end{align*}
  DaĂ­ que
  \begin{align*}
    e^{I}\wedge e^{J}(e_N)=&\frac{1}{k!l!}\sum_{\sigma\in S_{k+l}}(\sgn\sigma)e^{I}(e_{N^{k}_{\sigma}})e^{J}(e^{l}_{\sigma})\\
    =&\frac{1}{k!l!}\sum_{\substack{\tau\in S_{k}\\\eta\in S_{l}}}(\sgn\tau)(\sgn\eta)e^{I}(e_{N^{k}_{\tau}})e^{J}(e^{l}_{\eta})\\
    =&\left[ \frac{1}{k!}\sum_{\tau\in S_{k}}(\sgn\tau)e^{I}(e_{N^{k}_{\tau}}) \right]\left[ \frac{1}{l!}\sum_{\eta\in S_{l}}(\sgn\eta)e^{J}(e_{N^l_{\eta}}) \right]\\
    =&e^{I}(e_{N^{k}})e^{J}(e_{N^{l}})\\
    =&\delta_{N^{k}}^{I}\delta_{N^{l}}^{J}\\
    =&\delta_{N}^{IJ}\\
    =&e^{IJ}(e_{N}).
  \end{align*}
  Pelo que podemos concluir que
  \begin{align*}
    e^{I}\wedge e^{J}=e^{IJ}.
  \end{align*}
  Logo, fazendo um argumento inductivo temos que
  \begin{align*}
    e^{I}=e^{i_1}\wedge\cdots\wedge e^{i_k}.
  \end{align*}
\end{proof}

\begin{proposition}[Propriedade do produto exterior]
  O produto exterior satisfaz as seguintes propriedades
  \begin{enumerate}[i)]
    \item Se $a,b\in\mathbb{R}$, entĂŁo se cumpre que

      \end{align*}
    \item $(w\wedge\eta)\wedge\theta=w\wedge(\eta\wedge\theta)$.
    \item Se $w\in\Lambda^{k}(V)$ e $\eta\in\Lambda^{l}(V)$, entĂŁo
      \begin{align*}
        w\wedge\eta=(-1)^{kl}\eta\wedge w.
      \end{align*}

      \begin{align*}
        w^{1}\wedge\cdots\wedge w^{k}(v_1,\cdots,v_k)=\det\left[ w^{i}(v_j) \right].
      \end{align*}
      Em particular,
      \begin{align*}
        e^{1}\wedge\cdots\wedge e^{k}(v_1,\cdots,v_k)=\det\left[ v^{i}_{j} \right].
      \end{align*}
      Isto Ă©,
      \begin{align*}
        e^{1}\wedge\cdots\wedge e^{k}=\det.
      \end{align*}
  \end{enumerate}
\end{proposition}
\begin{proof}

    \item Pela definiĂ§ĂŁo.\\
    \item $(e^{I}\wedge e^{J})\wedge e^{K}=e^{IJ}\wedge e^{K}=e^{IJK}=\cdots$.
    \item $e^{I}\wedge e^{J}=e^{IJ}=(-1)^{kl}e^{JI}=(-1)^{kl}e^{J}\wedge e^{I}$.
    \item Sabemos que se $I,J$ sĂŁo multi-Ă­ndices de ordem $k$, entĂŁo 
      \begin{align*}
        e^{I}(e_J)=\det\left[ e^{i_r}(e_{j_s}) \right].

  \end{enumerate}
\end{proof}
