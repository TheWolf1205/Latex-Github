\begin{homeworkProblem}
  Seja $f : U \to \mathbb{R}^n$ Lipschitziana no aberto $U \subset \mathbb{R}^m$. Dado $a \in U$, suponha que, para todo $v \in \mathbb{R}^m$, existe a derivada direcional $\frac{\partial f}{\partial v}(a)$ e depende linearmente de $v$. Prove que, para todo caminho $g : (-\varepsilon, \varepsilon) \to U$, com $g(0) = a$, diferenciável no ponto $t = 0$, existe o vetor-velocidade $(f \circ g)'(0)$. Conclua que $f$ é diferenciável no ponto $a$.
  \begin{solution}
    Dado $a\in U$, defina $A(v)=\frac{\partial f}{\partial v}(a)$, logo ten-se que como $\frac{\partial f}{\partial v}(a)$ é linear en relação al vetor $v$, isso é equivalen a que $A(v)$ seja contínua en relação al vector $v\in\mathbb{R}^{m}$ (poiss $\mathbb{R}^{m}$ tem dimenção finita).\\
    Então, sabemos que como $g$ é diferenciável no ponto $t=0$ e $g(0)=a$ temos que para todo $t\in U$ se cumpre que 
    \begin{equation}\label{eq:11-1}
      g(t)=g(0+t)=g(0)+g^{\prime}(0)\cdot t+r(t)=a+g^{\prime}(0)\cdot t+r(t).
    \end{equation}
    Logo defina $h(t)=g(t)-a$, então
    \begin{equation}\label{eq:11-2}
      \frac{f(g(t))-f(g(0))}{t}=\frac{f(a+h(t))-f(a)}{t}=\frac{f(a+h(t))-f(a)}{|h(t)|}\cdot\frac{|h(t)|}{t}.
    \end{equation}
    Estudemos o vetor $v_t:=\frac{h(t)}{|t|}$, note que como $h(t)=g(t)-a$ por \eqref{eq:11-1} temos que $h(t)=g^{\prime}(0)\cdot t+r(t)$, logo $v_t\xrightarrow[t\to 0]{}g^{\prime}(0)$. Assim, usando que $f$ é Lipschitziana (suponha que a constante Lipschitziana é $K$) temos que 
    \begin{align*}
      \lim_{t \to 0}\left| \frac{f(a+tv_t)-f(a)}{t} - \frac{f(a+tg^{\prime}(0))-f(a)}{t}\right|&= \lim_{t \to 0}\left| \frac{f(a+tv_t)-f(a+tg^{\prime}(0))}{t} \right|\\
      &\leq \lim_{t \to 0}\frac{K|a+tv_t-a-tg^{\prime}(0)|}{t}\\
      &\leq \lim_{t \to 0}\frac{K|t||v_t-g^{\prime}(0)|}{|t|}\\
      &\leq \lim_{t \to 0}K|v_t-g^{\prime}(0)|\\
      &\leq 0.
    \end{align*}
    Logo temos que
    \begin{align*}
      \lim_{t \to 0}\frac{f(a+tv_t)-f(a)}{|t|}=\lim_{t \to 0}\frac{f(a+tg^{\prime}(0))-f(a)}{|t|}=A(g^{\prime}(0)).
    \end{align*}
    Agora estudemos e vetor velocidade $(f\circ g)^{\prime}(0)$, usando que $A$ é uma aplicação contínua e linear sabemos que pelo equação \eqref{eq:11-2} temos que
    \begin{equation}\label{eq:11-3}
      \lim_{t \to 0}\frac{f(g(t))-f(g(0))}{|t|}=\lim_{t \to 0}\frac{f(a+h(t))-f(a)}{|h(t)|}\cdot\frac{|h(t)|}{|t|}=A\left( \frac{g^{\prime}(0)}{|g^{\prime}(0)|} \right)\cdot |g^{\prime}(0)|=A(g^{\prime}(0)).
    \end{equation}
    Portanto, podemos definir $\frac{\partial f}{\partial v}(a)$ não so por segmentos (derivada direcional), se não por meio de caminhos.\\ 
    Note que se $g^{\prime}(0)=0$ temos, pelo teorema de valor medio que $g$ é constante, então 
    \begin{align*}
      \lim_{t \to 0}\frac{f(g(t))-f(g(0))}{|t|}=\lim_{t \to 0}\frac{0}{|t|}=0=A(g^{\prime}(0)).
    \end{align*}
    Portanto, temos que
    \begin{align*}
      (f\circ g)^{\prime}(0)=A(g^{\prime}(0)).
    \end{align*}
    Agora, definamos
    \begin{align*}
      r(v)=f(a+v)-f(a)-A(g^{\prime}(0))\cdot v.
    \end{align*}
    Assim, pela equação \eqref{eq:11-3} temos que
    \begin{align*}
      \lim_{v \to 0}\frac{r(v)}{|v|}=\lim_{v \to 0}\frac{f(a+v)-f(a)-A(g^{\prime}(0))\cdot v}{|v|}=\lim_{v \to 0}\frac{f(a+v)-f(a)}{|v|}-\frac{A(g^{\prime}(0))\cdot v}{|v|}=0.
    \end{align*}
    Portanto, podemos concluir que $f$ é diferenciável no ponto $a$. 
  \end{solution}
\end{homeworkProblem}
